\begin{frame}{Motivation}
\bit
  \item Initial PCP-specialist referrals shape downstream care
  \item Patients are increasingly reliant on referrals
  \item Most literature studies established networks; we study how those networks are \emph{formed}
\eit
\end{frame}

\begin{frame}{Empirical Setup}
\bit
  \item Medicare FFS data 2009--2018; elective orthopedic surgery referrals
  \item PCP defined by plurality of E\&M visits in prior year
  \item We focus on PCP movers to isolate initial referral decisions
  \item Network formation: any referral link within HRR in year of move 
\eit
\end{frame}

\begin{frame}{Estimation Strategy}
\bit
  \item Reduced-form network formation model (Jochmans 2018):
  \begin{align*}
  & u_{ij} = x_{ij}'\beta + \alpha_{i} + \alpha_{j} - \varepsilon_{ij} \\
  & \Pr(y_{ij} = 1 | x_{ij}) = \frac{\exp(x_{ij}'\beta + \alpha_{i} + \alpha_{j})}{1 + \exp(x_{ij}'\beta + \alpha_{i} + \alpha_{j})}
  \end{align*}
  \item Quadruple-based estimator eliminates two-sided fixed effects
\eit
\end{frame}

\begin{frame}{Estimation Strategy}
\bit
  \item For each quadruple $(i,k;j,l)$, transform outcome and covariates:
  \[
    \tilde{y}_m = \frac{(y_{ij} - y_{il}) - (y_{kj} - y_{kl})}{2}, \quad
    \tilde{x}_m = (x_{ij} - x_{il}) - (x_{kj} - x_{kl})
  \]
  \item Identification of $\beta$ comes entirely from discordant comparisons ($\tilde{y}_m \in \{-1,1\}$):
    \bit
      \item Two senders rank two receivers differently
      \item Sender and receiver fixed effects cancel
      \item $\implies$ Non-discordant quadruples yield no identifying variation and are excluded
    \eit
  \item Log-likelihood
  \[
    L(\beta) = \sum_{m=1}^M \mathbf{1}\{\tilde{y}_m = -1\} \log F(\tilde{x}_m'\beta) + 
    \mathbf{1}\{\tilde{y}_m = 1\} \log (1 - F(\tilde{x}_m'\beta))
  \]
\eit
\end{frame}

\begin{frame}{Conclusions}
\bit
  \item Initial referral networks shaped by practice ties and geographic proximity
  \item Demographic homophily plays a smaller initial role, but gender homophily still important
  \item Over time, practice ties, proximity, and demographic similarity effects attenuate, and effects of gender homophily dissipate entirely
  \item Reliance on practice affiliation has limited effects on patient quality
  \item Preference of nearby specialists may mediate effects of other potential biases
\eit
\end{frame}
